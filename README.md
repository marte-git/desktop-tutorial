/* n dimensione array, v valore intero */
int eser (int arr[], int n, int v) {
  int i, dist, distMin, indMin=0;

  distMin = (arr[0]<v) ? v-arr[0] : arr[0]-v;
  for(i=1; i<n; i++) {
     dist = (arr[i]<v) ? v-arr[i] : arr[i]-v;
     if (dist<distMin) {
        distMin = dist;
        indMin=i;
     }
  }
return indMin;
}
